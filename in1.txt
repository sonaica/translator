fun div(int n, int m) int {
    int i = 2;
    int cnt = 0;
    while (i * i <= n){
        if (n % i == 0) {
            output i, " ";
            cnt += 1;
            if (cnt == m) {
                return cnt;
            }
        }
        ++i;
    }
    return m;
}

    int n;
    input n;
    div(n, div(256, 3));
}
